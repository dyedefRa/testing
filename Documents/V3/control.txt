    private static string AuthenticationWithRestfullApi()
        {
            var request = (HttpWebRequest)WebRequest.Create("http://bip.boschevaletleri.com/api/account/login");
            //ServicePointManager.SecurityProtocol = SecurityProtocolType.Tls | (SecurityProtocolType)768 | (SecurityProtocolType)3072;
            //ServicePointManager.ServerCertificateValidationCallback = new
            //RemoteCertificateValidationCallback
            //(
            //   delegate { return true; }
            //);

            LoginRequestModel requestData = new LoginRequestModel
            {
                Username = ConfigurationSettings.AppSettings["UserName"],
                Password = ConfigurationSettings.AppSettings["Password"]
            };

            request.ContentType = "application/json";
            request.Method = "POST";
            var data = Encoding.ASCII.GetBytes(JsonConvert.SerializeObject(requestData));
            request.ContentLength = data.Length;
            using (var stream = request.GetRequestStream())
            {
                stream.Write(data, 0, data.Length);
            }

            var response = (HttpWebResponse)request.GetResponse();

            var responseString = new StreamReader(response.GetResponseStream()).ReadToEnd();
            var responseData = JsonConvert.DeserializeObject<LoginResponse>(responseString);
            return responseData.Data;
        }
