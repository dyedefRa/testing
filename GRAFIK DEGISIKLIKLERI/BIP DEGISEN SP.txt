ALTER PROCEDURE [dbo].[pa_Survey_List_ExcelExportDataQuestionsAnswersChoices]   
 @BrandId int,  
 @SurveyId int  
AS  
BEGIN  
 SELECT sqac.SurveyQuestionAnswerId, sqc.Choice, sqac.ChoiceRating, sqac.ChoiceComment  ,sqac.ChoiceOrderNumber
 FROM SurveyQuestionAnswerChoice sqac  
 INNER JOIN SurveyQuestionChoice sqc ON sqac.SurveyQuestionChoiceId = sqc.Id  
 INNER JOIN SurveyQuestionAnswer sqa ON sqa.Id = sqac.SurveyQuestionAnswerId  
 INNER JOIN SurveyQuestion sq ON sq.Id = sqa.SurveyQuestionId  
 INNER JOIN Survey s ON s.Id = sq.SurveyId  
 WHERE sqac.BrandId = @BrandId   
  AND s.Id = @SurveyId  
  AND sq.Status = 0  
END
